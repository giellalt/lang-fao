! The morphological tags
! ======================

! For each group, the maltag etc. lexicon functions as a default
! lexicon. The other lexica are there for specific subgroups of the names.


! Male first names
! ----------------

LEXICON maltag
+N+Prop+Nom: # ;
+N+Prop+Acc: # ;
+N+Prop+Dat:%>i # ;
+N+Prop+Gen:%>s # ;

LEXICON MUSTAFA
+N+Prop+Nom: # ;
+N+Prop+Acc: # ;
+N+Prop+Dat: # ;
+N+Prop+Gen: # ;
